    renderNav('patients', 'pp-ae', '../../../');
    let aeData = [];
    let currentSubject = null;
    // R code colors: sev <- c("Grade 1"="lightblue","Grade 2"="#42b83e","Grade 3"="orange","Grade 4"="#e00d10","Grade 5"="#570203")
    const severityColors = { 'Grade 1': 'lightblue', 'Grade 2': '#42b83e', 'Grade 3': 'orange', 'Grade 4': '#e00d10', 'Grade 5': '#570203' };
    // R code: fill <- c("Not Related"="black","Unlikely Related"="grey","Possibly Related"="red","Probably Related"="blue","Definitely Related"="green")
    const relationColors = { 'Not Related': 'black', 'Unlikely Related': 'grey', 'Possibly Related': 'red', 'Probably Related': 'blue', 'Definitely Related': 'green' };
    
    // R: safe_date <- function(x, fmt) as.Date(x, format = fmt)  with "%d %b %Y"
    function parseAEDate(str) {
      if (!str) return null;
      const parts = str.trim().split(' ');
      if (parts.length !== 3) return null;
      const months = {Jan:0,Feb:1,Mar:2,Apr:3,May:4,Jun:5,Jul:6,Aug:7,Sep:8,Oct:9,Nov:10,Dec:11};
      return new Date(parseInt(parts[2]), months[parts[1]], parseInt(parts[0]));
    }
    
    async function loadData() {
      return new Promise(resolve => {
        Papa.parse('../../../data/ae.csv', { download: true, header: true, skipEmptyLines: true, complete: r => { aeData = r.data; resolve(); }, error: () => resolve() });
      });
    }
    
    function populateSubjects() {
      const subjects = [...new Set(aeData.map(r => r.Subject))].filter(Boolean).sort();
      const select = document.getElementById('subjectSelect');
      select.innerHTML = subjects.map(s => `<option value="${s}">${s}</option>`).join('');
      const params = new URLSearchParams(window.location.search);
      currentSubject = params.get('id') || subjects[0];
      if (subjects.includes(currentSubject)) select.value = currentSubject;
      else currentSubject = subjects[0];
      select.addEventListener('change', e => { currentSubject = e.target.value; history.replaceState(null, '', `?id=${currentSubject}`); renderChart(); });
      renderChart();
    }
    
    function renderChart() {
      const container = document.getElementById('chartContainer');
      const subj = currentSubject;
      
      // Filter and process data matching R plot_ae_pref logic
      let data = aeData.filter(r => r.Subject === subj).map(r => ({
        ...r,
        startDate: parseAEDate(r.AESTDAT_RAW),
        endDate: parseAEDate(r.AEENDAT_RAW),
        // R: DisplayedTerm = if_else(is.na(AETERM_pt) | AETERM_pt == "", AETERM, AETERM_pt)
        displayTerm: (r.AETERM_pt && r.AETERM_pt.trim()) ? r.AETERM_pt : r.AETERM,
        // R: fallback = is.na(AETERM_pt) | AETERM_pt == ""
        isFallback: !r.AETERM_pt || r.AETERM_pt.trim() === ''
      })).filter(r => r.startDate);
      
      if (!data.length) { container.innerHTML = '<div class="no-data">No adverse events data available</div>'; return; }
      
      // R: start0 <- min(d$AESTDAT, na.rm = TRUE)
      const minDate = new Date(Math.min(...data.map(r => r.startDate.getTime())));
      // Calculate days from first AE
      data = data.map(r => ({
        ...r,
        startDay: Math.round((r.startDate - minDate) / (1000 * 60 * 60 * 24)),
        endDay: r.endDate ? Math.round((r.endDate - minDate) / (1000 * 60 * 60 * 24)) : null
      }));
      
      // R: ongoing <- max(c(d$EndDay, d$StartDay), na.rm = TRUE) + 5
      const maxDay = Math.max(...data.map(r => r.endDay !== null ? r.endDay : r.startDay));
      const ongoingDay = maxDay + 5;
      
      // R: d$EndDay <- if_else(is.na(d$AEENDAT), ongoing, d$EndDay)
      data = data.map(r => ({
        ...r,
        endDay: r.endDay === null ? ongoingDay : r.endDay,
        isOngoing: r.endDate === null
      }));
      
      // Group by term and sort
      const terms = [...new Set(data.map(r => r.displayTerm))];
      
      document.getElementById('pageTitle').textContent = `Adverse Events History for Subject ${subj}`;
      
      let html = `<div class="chart-title">Days since first AE at ${minDate.toISOString().split('T')[0]}</div>`;
      
      for (const term of terms) {
        const termData = data.filter(r => r.displayTerm === term).sort((a, b) => a.startDay - b.startDay);
        const isFallback = termData[0].isFallback;
        
        html += `<div class="ae-row">`;
        html += `<div class="ae-label${isFallback ? ' fallback' : ''}" title="${term}">${term}</div>`;
        html += `<div class="ae-bar-area">`;
        
        termData.forEach((ae, idx) => {
          const leftPct = (ae.startDay / ongoingDay) * 100;
          const widthPct = ((ae.endDay - ae.startDay) / ongoingDay) * 100;
          const sevColor = severityColors[ae.AETOXGR] || '#999';
          const relColor = relationColors[ae.AEREL] || 'black';
          // R: odd = idx %% 2 == 1 for alternating label position
          const odd = idx % 2 === 0;
          const dateLabel = `${(ae.startDate.getMonth()+1).toString().padStart(2,'0')}.${ae.startDate.getDate().toString().padStart(2,'0')}`;
          
          // Segment (bar)
          if (ae.startDay !== ae.endDay) {
            html += `<div class="ae-segment" style="left:${leftPct}%;width:${widthPct}%;background:${sevColor};"></div>`;
          }
          
          // Point at start
          html += `<div class="ae-point" style="left:${leftPct}%;background:${relColor};${ae.startDay === ae.endDay ? 'border:2px solid '+sevColor+';' : ''}"></div>`;
          
          // Arrow for ongoing or end mark
          if (ae.isOngoing) {
            html += `<div class="ae-arrow" style="left:calc(${leftPct + widthPct}% - 8px);color:${sevColor};">→</div>`;
          } else if (ae.startDay !== ae.endDay) {
            html += `<div class="ae-end-mark" style="left:${leftPct + widthPct}%;">|</div>`;
          }
          
          // Date label with alternating position
          html += `<div class="ae-date-label" style="left:${leftPct}%;top:${odd ? '-12px' : '18px'};">${dateLabel}</div>`;
        });
        
        html += `</div></div>`;
      }
      
      // X-axis
      html += '<div class="x-axis">';
      const step = Math.ceil(ongoingDay / 10);
      for (let d = 0; d <= ongoingDay; d += step) {
        html += `<span>${d === ongoingDay ? 'Ongoing' : d}</span>`;
      }
      html += '</div>';
      
      container.innerHTML = html;
    }
    
    loadData().then(populateSubjects);
  </script>

    function parseAEDate(str) {
      if (!str) return null;
      const parts = str.trim().split(' ');
      if (parts.length !== 3) return null;
      const months = {Jan:0,Feb:1,Mar:2,Apr:3,May:4,Jun:5,Jul:6,Aug:7,Sep:8,Oct:9,Nov:10,Dec:11};
      return new Date(parseInt(parts[2]), months[parts[1]], parseInt(parts[0]));
    }
    function parseSasDate(str) {
      if (!str) return null;
      const dateOnly = str.substring(0, 9);
      const months = {JAN:0,FEB:1,MAR:2,APR:3,MAY:4,JUN:5,JUL:6,AUG:7,SEP:8,OCT:9,NOV:10,DEC:11};
      return new Date(parseInt(dateOnly.substring(5,9)), months[dateOnly.substring(2,5).toUpperCase()], parseInt(dateOnly.substring(0,2)));
    }

    function renderChart() {
      const container = document.getElementById('chartContainer');
      const subj = currentSubject;
      
      // Get C1D1 date from visit data - R: c1 <- v %>% filter(Folder == "C1D1")
      const visSubj = visData.filter(r => r.Subject === subj);
      const c1d1Row = visSubj.find(r => r.Folder === 'C1D1');
      if (!c1d1Row) { container.innerHTML = '<div class="no-data">No C1D1 visit data available</div>'; return; }
      const c1d1Date = parseSasDate(c1d1Row.VISDAT);
      if (!c1d1Date) { container.innerHTML = '<div class="no-data">Invalid C1D1 date</div>'; return; }
      
      let data = aeData.filter(r => r.Subject === subj).map(r => ({
        ...r,
        startDate: parseAEDate(r.AESTDAT_RAW),
        endDate: parseAEDate(r.AEENDAT_RAW),
        displayTerm: (r.AETERM_pt && r.AETERM_pt.trim()) ? r.AETERM_pt : r.AETERM,
        isFallback: !r.AETERM_pt || r.AETERM_pt.trim() === ''
      })).filter(r => r.startDate);
      
      if (!data.length) { container.innerHTML = '<div class="no-data">No adverse events data</div>'; return; }
      
      // R: StartDay = as.numeric(difftime(AESTDAT, c1d1, units = "days"))

        ...r,
        startDay: Math.round((r.startDate - c1d1Date) / (1000 * 60 * 60 * 24)),
        endDay: r.endDate ? Math.round((r.endDate - c1d1Date) / (1000 * 60 * 60 * 24)) : null
      }));
      
      // R: Pre_day <- -30
      const PRE_DAY = -30;
      // R: ongoing <- max(c(d$EndDay, d$StartDay), na.rm = TRUE) + 30
      const maxDay = Math.max(...data.map(r => r.endDay !== null ? r.endDay : r.startDay));
      const ongoingDay = maxDay + 30;
      
      data = data.map(r => ({ ...r, endDay: r.endDay === null ? ongoingDay : r.endDay, isOngoing: r.endDate === null }));
      
      // Get visit labels for dual axis
      const visits = visSubj.map(r => ({

      })).filter(v => !isNaN(v.day)).sort((a, b) => a.day - b.day);
      
      const axisBreaks = [PRE_DAY, ...visits.map(v => v.day), ongoingDay];
      const axisLabelsTop = ['Pre', ...visits.map(v => v.day), 'Ongoing'];
      const axisLabelsBot = ['Pre', ...visits.map(v => v.folder), 'Ongoing'];
      
      const terms = [...new Set(data.map(r => r.displayTerm))];
      const range = ongoingDay - PRE_DAY;

        html += `<span style="position:absolute;left:calc(200px + ${pct}%);transform:translateX(-50%);">${axisLabelsTop[i]}</span>`;
      });

      // Data rows
      for (const term of terms) {

        html += `<div class="ae-row"><div class="ae-label${termData[0].isFallback ? ' fallback' : ''}">${term}</div><div class="ae-bar-area">`;
        
        for (const ae of termData) {
          const leftPct = ((ae.startDay - PRE_DAY) / range) * 100;
          const widthPct = ((ae.endDay - ae.startDay) / range) * 100;
          const sevColor = severityColors[ae.AETOXGR] || '#999';
          const relColor = relationColors[ae.AEREL] || 'black';
          const duration = ae.endDay - ae.startDay || 1;
          
          if (ae.startDay !== ae.endDay) {
            html += `<div class="ae-segment" style="left:${leftPct}%;width:${widthPct}%;background:${sevColor};"></div>`;
          }
          html += `<div class="ae-point" style="left:${leftPct}%;background:${relColor};"></div>`;
          
          if (ae.isOngoing) {
            html += `<span style="position:absolute;left:calc(${leftPct + widthPct}% - 8px);top:8px;font-size:14px;color:${sevColor};">→</span>`;
          } else if (ae.startDay !== ae.endDay) {
            html += `<span style="position:absolute;left:${leftPct + widthPct}%;top:6px;font-size:14px;">|</span>`;
          }

        html += `<span style="position:absolute;left:calc(200px + ${pct}%);transform:translateX(-50%);">${axisLabelsBot[i]}</span>`;
      });
      html += '</div></div>';
      
      container.innerHTML = html;
