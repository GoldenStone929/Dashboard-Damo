    let cmData = [], visData = [];
    let currentSubject = null;
    const PRE_DAY = -30; // R: Pre_day <- -30
    
    function parseCmDate(str) {
      if (!str) return null;
      const dateOnly = str.substring(0, 9);
      const months = {JAN:0,FEB:1,MAR:2,APR:3,MAY:4,JUN:5,JUL:6,AUG:7,SEP:8,OCT:9,NOV:10,DEC:11};
      const day = parseInt(dateOnly.substring(0,2));
      const mon = months[dateOnly.substring(2,5).toUpperCase()];
      const year = parseInt(dateOnly.substring(5,9));
      if (isNaN(day) || mon === undefined || isNaN(year)) return null;
      return new Date(year, mon, day);
    }
    
    async function loadData() {
      const [cm, vis] = await Promise.all([
        new Promise(res => Papa.parse('../../../data/cm.csv', { download: true, header: true, skipEmptyLines: true, complete: r => res(r.data), error: () => res([]) })),
        new Promise(res => Papa.parse('../../../data/vis.csv', { download: true, header: true, skipEmptyLines: true, complete: r => res(r.data), error: () => res([]) }))
      ]);
      cmData = cm; visData = vis;

    function populateSubjects() {
      const subjects = [...new Set(cmData.map(r => r.Subject))].filter(Boolean).sort();
      const select = document.getElementById('subjectSelect');
      select.innerHTML = subjects.map(s => `<option value="${s}">${s}</option>`).join('');
      const params = new URLSearchParams(window.location.search);
      currentSubject = params.get('id') || subjects[0];
      if (subjects.includes(currentSubject)) select.value = currentSubject;

      select.addEventListener('change', e => { currentSubject = e.target.value; history.replaceState(null, '', `?id=${currentSubject}`); renderChart(); });

    }
    
    function renderChart() {
      const container = document.getElementById('chartContainer');
      const subj = currentSubject;
      
      const visSubj = visData.filter(r => r.Subject === subj);
      const c1d1Row = visSubj.find(r => r.Folder === 'C1D1');
      if (!c1d1Row) { container.innerHTML = '<div class="no-data">No C1D1 visit data</div>'; return; }
      const c1d1Date = parseCmDate(c1d1Row.VISDAT);
      
      // Get visit axis labels
      const visits = visSubj.map(r => ({
        folder: r.Folder === 'DSET' ? 'EOT' : r.Folder,
        day: Math.round((parseCmDate(r.VISDAT) - c1d1Date) / (1000*60*60*24))
      })).filter(v => !isNaN(v.day)).sort((a, b) => a.day - b.day);
      
      // R: filter(CM_StudyDay_Start >= -30)
      let data = cmData.filter(r => r.Subject === subj).map(r => {
        const startDate = parseCmDate(r.CMSTDAT_INT);
        const endDate = parseCmDate(r.CMENDAT_INT);
        const startDay = startDate ? Math.round((startDate - c1d1Date) / (1000*60*60*24)) : null;
        const endDay = endDate ? Math.round((endDate - c1d1Date) / (1000*60*60*24)) : null;
        const medication = [r.CMTRT, r.CMDSTXT, r.CMDOSU_STD, r.CMROUTE_STD, r.CMDOSFRQ_STD].filter(Boolean).join(' - ');
        return { ...r, startDate, endDate, startDay, endDay, medication, status: endDate ? 'Complete' : 'Started Only' };
      }).filter(r => r.startDay !== null && r.startDay >= PRE_DAY);
      
      if (!data.length) { container.innerHTML = '<div class="no-data">No concomitant medications data (≥ -30 days)</div>'; return; }
      
      // R: ongoing <- max(c(d$CM_StudyDay_Start, d$CM_StudyDay_End), na.rm = TRUE) + 14
      const maxDay = Math.max(...data.map(r => r.endDay !== null ? r.endDay : r.startDay));
      const ongoingDay = maxDay + 14;
      const range = ongoingDay - PRE_DAY;
      
      data = data.map(r => ({ ...r, endDay: r.endDay === null ? ongoingDay : r.endDay }));
      
      const axisBreaks = [PRE_DAY, ...visits.map(v => v.day), ongoingDay];
      const axisLabelsTop = ['Pre', ...visits.map(v => v.day), 'Ongoing'];
      const axisLabelsBot = ['Pre', ...visits.map(v => v.folder), 'Ongoing'];
      
      let html = '<div class="dual-axis">';
      
      // Top axis
      html += '<div class="axis-row top">';
      axisBreaks.forEach((b, i) => {
        const pct = ((b - PRE_DAY) / range) * 100;
        html += `<span style="position:absolute;left:${pct}%;transform:translateX(-50%);">${axisLabelsTop[i]}</span>`;
      });
      html += '</div>';
      
      // Data rows
      for (const cm of data) {
        const leftPct = ((cm.startDay - PRE_DAY) / range) * 100;
        const widthPct = ((cm.endDay - cm.startDay) / range) * 100;
        const color = cm.status === 'Complete' ? 'black' : 'red';
        const dateLabel = cm.startDate ? cm.startDate.toISOString().split('T')[0].replace(/-/g, '.') : '';
        
        html += `<div class="cm-row"><div class="cm-label" title="${cm.medication}">${cm.medication}</div><div class="cm-bar-area">`;
        html += `<div class="cm-segment" style="left:${leftPct}%;width:${widthPct}%;background:${color};"></div>`;
        html += `<div class="cm-point" style="left:${leftPct}%;background:${color};"></div>`;
        if (cm.status === 'Complete') {
          html += `<div class="cm-point" style="left:${leftPct + widthPct}%;background:${color};"></div>`;
        } else {
          html += `<span style="position:absolute;left:calc(${leftPct + widthPct}% - 5px);top:5px;font-size:14px;color:red;">→</span>`;
        }
        html += `<div class="cm-date-label" style="left:${leftPct}%;">${dateLabel}</div>`;
        html += '</div></div>';
      }
      
      // Bottom axis
      html += '<div class="axis-row bottom">';
      axisBreaks.forEach((b, i) => {
        const pct = ((b - PRE_DAY) / range) * 100;
        html += `<span style="position:absolute;left:${pct}%;transform:translateX(-50%);">${axisLabelsBot[i]}</span>`;
      });
      html += '</div></div>';
      
      container.innerHTML = html;
    }

    let cmData = [];
    let currentSubject = null, currentPage = 1;
    const ROWS_PER_PAGE = 15; // R: rows_per_page <- 15L
    
    function shortDate(str) { return (!str || str.trim() === '') ? '' : str.trim().substring(0, 9); }
    function parseCmDate(str) {
      if (!str) return null;
      const dateOnly = str.substring(0, 9);
      const months = {JAN:0,FEB:1,MAR:2,APR:3,MAY:4,JUN:5,JUL:6,AUG:7,SEP:8,OCT:9,NOV:10,DEC:11};
      const day = parseInt(dateOnly.substring(0,2)), mon = months[dateOnly.substring(2,5).toUpperCase()], year = parseInt(dateOnly.substring(5,9));
      if (isNaN(day) || mon === undefined || isNaN(year)) return null;
      return new Date(year, mon, day);
    }
    
    async function loadData() {
      return new Promise(res => Papa.parse('../../../data/cm.csv', { download: true, header: true, skipEmptyLines: true, complete: r => { cmData = r.data; res(); }, error: () => res() }));
    }
    
    function populateSubjects() {
      const subjects = [...new Set(cmData.map(r => r.Subject))].filter(Boolean).sort();
      const select = document.getElementById('subjectSelect');
      select.innerHTML = subjects.map(s => `<option value="${s}">${s}</option>`).join('');
      const params = new URLSearchParams(window.location.search);
      currentSubject = params.get('id') || subjects[0];
      if (subjects.includes(currentSubject)) select.value = currentSubject;

      renderTable();
    }
    
    function renderTable() {
      const container = document.getElementById('tableContainer');
      const subj = currentSubject;
      
      // R transformation: matching tbl structure
      let data = cmData.filter(r => r.Subject === subj).map(r => {
        const startDate = parseCmDate(r.CMSTDAT_INT);
        const endDate = parseCmDate(r.CMENDAT_INT);
        const studyDays = (startDate && endDate) ? Math.round((endDate - startDate) / (1000*60*60*24)) : '';
        return {
          medication: r.CMTRT || '',
          category: r.CMCAT || '',
          reason: r.CMREAS || '',
          dose: r.CMDSTXT || '',
          unit: r.CMDOSU_STD || r.CMDOSU || '',
          frequency: r.CMDOSFRQ_STD || r.CMDOSFRQ || '',
          route: r.CMROUTE_STD || r.CMROUTE || '',
          startDate: shortDate(r.CMSTDAT_INT),
          endDate: shortDate(r.CMENDAT_INT),
          ongoing: r.CMONGO || '',
          studyDays: studyDays,
          aeRelated: r.CMAENO_STD || r.CMAENO || '',
          _sortDate: startDate
        };
      }).sort((a, b) => (a._sortDate || 0) - (b._sortDate || 0));
      
      if (!data.length) { container.innerHTML = '<div class="no-data">No concomitant medications data</div>'; document.getElementById('pagination').innerHTML = ''; return; }
      
      const totalPages = Math.ceil(data.length / ROWS_PER_PAGE);
      const startIdx = (currentPage - 1) * ROWS_PER_PAGE;
      const pageData = data.slice(startIdx, startIdx + ROWS_PER_PAGE);
      
      document.getElementById('pageTitle').textContent = `Concomitant Medications Data Listing${totalPages > 1 ? ` (Page ${currentPage}/${totalPages})` : ''}`;
      
      // R column widths: c(0.12, 0.08, 0.10, 0.06, 0.05, 0.08, 0.06, 0.09, 0.09, 0.07, 0.08, 0.10)
      let html = `<table class="cm-table"><thead><tr>
        <th style="width:12%">Medication</th><th style="width:8%">Category</th><th style="width:10%">Reason</th>
        <th style="width:6%">Dose</th><th style="width:5%">Unit</th><th style="width:8%">Frequency</th>
        <th style="width:6%">Route</th><th style="width:9%">Start Date</th><th style="width:9%">End Date</th>
        <th style="width:7%">Ongoing</th><th style="width:8%">Study Days</th><th style="width:10%">AE Related</th>
      </tr></thead><tbody>`;
      for (const row of pageData) {
        html += `<tr><td>${row.medication}</td><td>${row.category}</td><td>${row.reason}</td><td>${row.dose}</td><td>${row.unit}</td><td>${row.frequency}</td><td>${row.route}</td><td>${row.startDate}</td><td>${row.endDate}</td><td>${row.ongoing}</td><td>${row.studyDays}</td><td>${row.aeRelated}</td></tr>`;
      }
      html += '</tbody></table>';
      container.innerHTML = html;
      
      let pagHtml = '';
      if (totalPages > 1) {
        pagHtml += `<button ${currentPage === 1 ? 'disabled' : ''} onclick="goToPage(${currentPage - 1})">← Prev</button>`;
        for (let p = 1; p <= totalPages; p++) pagHtml += `<button class="${p === currentPage ? 'active' : ''}" onclick="goToPage(${p})">${p}</button>`;
        pagHtml += `<button ${currentPage === totalPages ? 'disabled' : ''} onclick="goToPage(${currentPage + 1})">Next →</button>`;
      }
      document.getElementById('pagination').innerHTML = pagHtml;
    }
